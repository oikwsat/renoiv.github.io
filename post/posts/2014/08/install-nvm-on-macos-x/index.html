install nvm on MacOS X

install nvm

    $ brew install nvm
    $ source $(brew --prefix nvm)/nvm.sh
    $ export NVM_DIR=~/.nvm

    $ vim ~/.bash_profile
    source $(brew --prefix nvm)/nvm.sh
    export NVM_DIR=~/.nvm

install node.js

    $ nvm intall v0.10.26
    $ nvm intall v0.10.30
    $ nvm ls
        v0.10.26
    ->  v0.10.30
        system
    $ node -v
    v0.10.30

switch version

    $ nvm use v0.10.26
    $ node -v
    v0.10.26
